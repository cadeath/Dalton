SELECT 
  P.PAWNTICKET, P.LOANDATE, P.EXPIRYDATE, P.AUCTIONDATE, 
  C.FIRSTNAME || ' ' || C.LASTNAME || ' ' || C.SUFFIX AS CLIENT,
    C.ADDR_STREET || ' ' || C.ADDR_CITY || ' ' || C.ADDR_CITY || ' ' || C.ADDR_ZIP as FULLADDRESS,
    ITM.ITEMCLASS, CLASS.ITEMCATEGORY, LIST(PI1.SPECSVALUE || PI1.UOM || ' '),
    P.OLDTICKET, P.ORNUM, P.ORDATE, P.PRINCIPAL, 
    P.NETAMOUNT, P.RENEWDUE, P.REDEEMDUE, P.APPRAISAL,
    ITM.STATUS
FROM OPT P
  INNER JOIN TBLCLIENT C
  ON P.CLIENTID = C.CLIENTID
  INNER JOIN OPI ITM
    ON ITM.PAWNITEMID = P.PAWNITEMID
  INNER JOIN PI1
    ON PI1.PAWNITEMID = ITM.PAWNITEMID
  INNER JOIN TBLITEM CLASS
    ON CLASS.ITEMID = ITM.ITEMID
GROUP BY 
  P.PAWNTICKET, P.LOANDATE, P.EXPIRYDATE, P.AUCTIONDATE, 
    C.FIRSTNAME || ' ' || C.LASTNAME || ' ' || C.SUFFIX, 
    C.ADDR_STREET || ' ' || C.ADDR_CITY || ' ' || C.ADDR_CITY || ' ' || C.ADDR_ZIP,
    ITM.ITEMCLASS, CLASS.ITEMCATEGORY,
    P.OLDTICKET, P.ORNUM, P.ORDATE, P.PRINCIPAL, 
    P.NETAMOUNT, P.RENEWDUE, P.REDEEMDUE, P.APPRAISAL,
    ITM.STATUS


include Appraiser, PullOut Date, Encoder
